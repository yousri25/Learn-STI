<script>
function calculer() {
  let n1 = Number(document.getElementById("maths").value);
  let n2 = Number(document.getElementById("fr").value);
  let n3 = Number(document.getElementById("info").value);
  let moy = (n1 + n2 + n3) / 3;

  let mention;
  if      (moy >= 16) mention = "Très bien 🏆";
  else if (moy >= 14) mention = "Bien ⭐";
  else if (moy >= 12) mention = "Assez bien ✅";
  else if (moy >= 10) mention = "Passable";
  else                mention = "Insuffisant ❌";

  let el = document.getElementById("resultat");
  el.style.display = "block";
  el.innerHTML = "Moyenne : " + moy.toFixed(2) + "/20 — " + mention;
}

Un formulaire d'inscription avec validation JavaScript côté client : champs obligatoires, email, mot de passe confirmé.

  • Afficher les messages d'erreur en rouge ou un message de succès
<form onsubmit="return valider()">
  <input type="text"     id="nom"   placeholder="Nom complet">
  <input type="email"    id="email" placeholder="Email">
  <input type="password" id="pass1" placeholder="Mot de passe">
  <input type="password" id="pass2" placeholder="Confirmer">
  <div id="msg"></div>
  <button type="submit">S'inscrire</button>
</form>

<script>
function valider() {
  let nom   = document.getElementById("nom").value.trim();
  let email = document.getElementById("email").value.trim();
  let p1    = document.getElementById("pass1").value;
  let p2    = document.getElementById("pass2").value;
  let msg   = document.getElementById("msg");

  if (nom === "" || email === "" || p1 === "") {
    msg.innerHTML = "❌ Tous les champs sont obligatoires.";
    msg.style.color = "red";
    return false;
  }
  if (p1 !== p2) {
    msg.innerHTML = "❌ Les mots de passe ne correspondent pas.";
    msg.style.color = "red";
    return false;
  }
  msg.innerHTML = "✅ Inscription réussie ! Bienvenue " + nom;
  msg.style.color = "green";
  return false; // empêche rechargement pour démo
}
</script>

Base de données Bibliothèque
Conçois et crée une base de données pour gérer les livres et auteurs d'une bibliothèque.
Moyen
CREATE TABLE FOREIGN KEY INSERT INTO SELECT + JOIN
Objectifs
  • Créer deux tables liées : auteur et livre
  • Définir les types, contraintes et clé étrangère
  • Insérer des enregistrements de test
  • Écrire des requêtes SELECT avec jointure, filtre et tri
-- Créer la base
CREATE DATABASE bibliotheque;

-- Table auteur
CREATE TABLE auteur (
  id_auteur  INT PRIMARY KEY AUTO_INCREMENT,
  nom        VARCHAR(80)  NOT NULL,
  nationalite VARCHAR(50)
);

-- Table livre
CREATE TABLE livre (
  id_livre   INT PRIMARY KEY AUTO_INCREMENT,
  titre      VARCHAR(150) NOT NULL,
  annee      INT,
  genre      VARCHAR(50),
  id_auteur  INT,
  FOREIGN KEY (id_auteur)
    REFERENCES auteur(id_auteur)
    ON DELETE CASCADE
);

  ('Mahmoud Darwich',   'Palestinien');

INSERT INTO livre (titre, annee, genre, id_auteur) VALUES
  ('La Nuit Sacrée',    1987, 'Roman',  1),
  ('L\'Étranger',        1942, 'Roman',  2),
  ('La Peste',          1947, 'Roman',  2),
  ('Journal d\'un arbre', 1973, 'Poésie', 3);
-- Tous les livres avec le nom de l'auteur
SELECT   l.titre, l.annee, a.nom
FROM     livre AS l, auteur AS a
WHERE    l.id_auteur = a.id_auteur
ORDER BY l.annee ASC;

-- Livres d'Albert Camus
SELECT   l.titre, l.annee
FROM     livre AS l, auteur AS a
WHERE    l.id_auteur = a.id_auteur
AND      a.nom = 'Albert Camus';

-- Nombre de livres par auteur
SELECT   a.nom, COUNT(l.id_livre) AS "Nb livres"
FROM     livre AS l, auteur AS a
WHERE    l.id_auteur = a.id_auteur
GROUP BY a.nom;

Gestion des élèves
Base de données scolaire : classes, élèves et notes avec requêtes de statistiques par classe.
Difficile
Relations PK/FK AVG / MAX / MIN GROUP BY + HAVING ORDER BY
Objectifs
  • Créer 3 tables liées : classe, eleve, note
  • Calculer la moyenne, le max et le min par classe
  • Lister les élèves admis (moyenne ≥ 10)
  • Trouver les classes avec une moyenne générale ≥ 12
CREATE TABLE classe (
  id_classe INT PRIMARY KEY AUTO_INCREMENT,
  libelle   VARCHAR(30) NOT NULL
);

CREATE TABLE eleve (
  id_eleve  INT PRIMARY KEY AUTO_INCREMENT,
  nom       VARCHAR(50) NOT NULL,
  prenom    VARCHAR(50) NOT NULL,
  id_classe INT,
  FOREIGN KEY (id_classe) REFERENCES classe(id_classe)
);

CREATE TABLE note (
  id_note   INT PRIMARY KEY AUTO_INCREMENT,
  matiere   VARCHAR(50),
  valeur    DECIMAL(4,2) CHECK (valeur BETWEEN 0 AND 20),
  id_eleve  INT,
  FOREIGN KEY (id_eleve) REFERENCES eleve(id_eleve)
);
-- Moyenne par élève
SELECT   e.nom, e.prenom, AVG(n.valeur) AS "Moyenne"
FROM     eleve e, note n
WHERE    e.id_eleve = n.id_eleve
GROUP BY e.id_eleve
HAVING   AVG(n.valeur) >= 10
ORDER BY AVG(n.valeur) DESC;

-- Stats par classe
SELECT   c.libelle,
         AVG(n.valeur) AS "Moy",
         MAX(n.valeur) AS "Max",
         MIN(n.valeur) AS "Min"
FROM     classe c, eleve e, note n
WHERE    c.id_classe = e.id_classe
AND      e.id_eleve  = n.id_eleve
GROUP BY c.id_classe
HAVING   AVG(n.valeur) >= 12;

Livre d'or (Guestbook)
Un formulaire HTML envoie un message en PHP qui l'enregistre en MySQL. La page affiche ensuite tous les messages.
Moyen
$_POST mysqli_connect INSERT + SELECT mysqli_fetch_array
Objectifs
  • Créer la table message(id, nom, texte, date_envoi)
  • Créer un formulaire HTML avec méthode POST
  • PHP reçoit $_POST, insère en base avec mysqli
  • PHP lit tous les messages et les affiche dans la page
CREATE DATABASE guestbook;

CREATE TABLE message (
  id         INT PRIMARY KEY AUTO_INCREMENT,
  nom        VARCHAR(80)  NOT NULL,
  texte      TEXT         NOT NULL,
  date_envoi DATETIME DEFAULT NOW()
);
<?php
// Connexion à la base de données
$conn = mysqli_connect("localhost", "root", "", "guestbook");

// Traitement du formulaire (POST)
if (isset($_POST["nom"])) {
  $nom   = $_POST["nom"];
  $texte = $_POST["texte"];
  $sql   = "INSERT INTO message (nom, texte) VALUES ('$nom', '$texte')";
  mysqli_query($conn, $sql);
}
?>
<!DOCTYPE html><html lang="fr">
<head><meta charset="UTF-8"><title>Livre d'or</title></head>
<body>
  <h1>📖 Livre d'or</h1>

  <!-- Formulaire -->
  <form method="post">
    <input type="text"     name="nom"   placeholder="Votre nom"     required>
    <textarea              name="texte" placeholder="Votre message" required></textarea>
    <input type="submit" value="Envoyer">
  </form>

  <h2>Messages</h2>
  <?php
  // Lire et afficher les messages
  $res = mysqli_query($conn, "SELECT * FROM message ORDER BY date_envoi DESC");
  while ($row = mysqli_fetch_array($res)) {
    echo "<div><strong>" . $row["nom"] . "</strong> — "
       . $row["texte"]
       . " <small>(" . $row["date_envoi"] . "</small></div>";
  }
  mysqli_close($conn);
  ?>
</body></html>

CRUD Étudiants
Application complète de gestion d'étudiants : ajouter, afficher dans un tableau, et supprimer des enregistrements via PHP et MySQL.
Difficile
CRUD complet $_GET / $_POST Tableau HTML dynamique DELETE avec id
Objectifs
  • Créer la table etudiant(id, nom, prenom, note)
  • Formulaire HTML POST → PHP insère en base
  • PHP lit tous les étudiants et génère un tableau HTML
  • Lien "Supprimer" passe l'id en GET → PHP exécute DELETE
CREATE DATABASE gestion_etudiants;

CREATE TABLE etudiant (
  id     INT PRIMARY KEY AUTO_INCREMENT,
  nom    VARCHAR(60)  NOT NULL,
  prenom VARCHAR(60)  NOT NULL,
  note   DECIMAL(4,2) CHECK (note BETWEEN 0 AND 20)
);
<?php
$conn = mysqli_connect("localhost", "root", "", "gestion_etudiants");

// Ajout
if (isset($_POST["nom"])) {
  $n = $_POST["nom"]; $p = $_POST["prenom"]; $nt = $_POST["note"];
  mysqli_query($conn, "INSERT INTO etudiant (nom,prenom,note) VALUES ('$n','$p','$nt')");
}

// Suppression
if (isset($_GET["suppr"])) {
  $id = $_GET["suppr"];
  mysqli_query($conn, "DELETE FROM etudiant WHERE id = $id");
}
?>
